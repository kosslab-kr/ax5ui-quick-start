Make Button

<button id="mask-open" type="button">Alert</button>

Basic Alert Script

<script>
    $(document.body).ready(function () {
        var dialog = new ax5.ui.dialog({
            title: "AX5 Dialog"
        });

        $('#alert-open').click(function () {
            dialog.alert('Alert message');
        });
        $('#alert-close').click(function () {
            dialog.close();
        });
    });
</script>
